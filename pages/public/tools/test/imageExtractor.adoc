== Image Extractor

Ein verstecktes `<video>`-Element lädt die Datei, wird per `currentTime`
auf die gewünschte Position gesprungen, und sobald `onseeked` feuert,
zeichnet `canvas.drawImage(video, ...)` den aktuellen Frame ins Canvas.

Danach liefert `canvas.toDataURL()` das fertige Bild als Base64-String.

<script>
let vid = document.getElementById('vid');
let cnv = document.getElementById('cnv');
let ctx = cnv.getContext('2d');
let thumbCount = 0;

function switchTab(t) {
  document.getElementById('panel-file').style.display = t==='file'?'':'none';
  document.getElementById('panel-url').style.display = t==='url'?'':'none';
  document.getElementById('tab-file').className = 'vte-tab'+(t==='file'?' active':'');
  document.getElementById('tab-url').className = 'vte-tab'+(t==='url'?' active':'');
}

function onDrag(e, enter) {
  e.preventDefault();
  document.getElementById('dropzone').classList.toggle('drag-over', enter);
}

function onDrop(e) {
  e.preventDefault();
  document.getElementById('dropzone').classList.remove('drag-over');
  const f = e.dataTransfer.files[0];
  if (f && f.type.startsWith('video/')) setupVideo(URL.createObjectURL(f), f.name);
}

function loadFile(input) {
  const f = input.files[0];
  if (f) setupVideo(URL.createObjectURL(f), f.name);
}

function loadUrl() {
  const u = document.getElementById('urlInput').value.trim();
  if (u) setupVideo(u, u.split('/').pop() || 'video');
}

function setupVideo(src, name) {
  vid.src = src;
  document.getElementById('fname').textContent = name;
  vid.onloadedmetadata = () => {
    const dur = vid.duration;
    const seek = document.getElementById('seek');
    seek.max = dur;
    seek.step = dur / 1000;
    document.getElementById('vdim').textContent = vid.videoWidth + ' × ' + vid.videoHeight + ' px · ' + formatTime(dur);
    document.getElementById('controls').classList.add('show');
    seekTo(dur * 0.1);
  };
}

function onSeek(el) {
  seekTo(parseFloat(el.value));
}

function seekTo(t) {
  vid.currentTime = t;
  document.getElementById('seek').value = t;
  document.getElementById('seektime').textContent = formatTime(t);
  vid.onseeked = snapPreview;
}

function snapPreview() {}

function captureFrame() {
  const w = parseInt(document.getElementById('outW').value) || 320;
  const ratio = vid.videoHeight / vid.videoWidth;
  const h = Math.round(w * ratio);
  cnv.width = w;
  cnv.height = h;
  ctx.drawImage(vid, 0, 0, w, h);
  const fmt = document.getElementById('fmt').value;
  const qual = parseFloat(document.getElementById('qual').value);
  return { dataUrl: cnv.toDataURL(fmt, qual), time: vid.currentTime, w, h, fmt };
}

function snapCurrent() {
  const {dataUrl, time, w, h, fmt} = captureFrame();
  addThumb(dataUrl, time, w, h, fmt);
}

function snapMultiple() {
  const dur = vid.duration;
  if (!dur) return;
  const times = [0.05, 0.15, 0.30, 0.50, 0.70, 0.90].map(f => f * dur);
  let i = 0;
  function next() {
    if (i >= times.length) return;
    vid.currentTime = times[i];
    vid.onseeked = () => {
      const {dataUrl, time, w, h, fmt} = captureFrame();
      addThumb(dataUrl, time, w, h, fmt);
      i++;
      next();
    };
  }
  next();
}

function addThumb(dataUrl, time, w, h, fmt) {
  thumbCount++;
  const ext = fmt.split('/')[1];
  const grid = document.getElementById('thumbGrid');
  const div = document.createElement('div');
  div.className = 'vte-thumb';
  const img = document.createElement('img');
  img.src = dataUrl;
  img.alt = 'Frame @ ' + formatTime(time);
  const tspan = document.createElement('div');
  tspan.className = 'vte-time';
  tspan.textContent = formatTime(time) + ' · ' + w + '×' + h;
  const a = document.createElement('a');
  a.className = 'vte-dl';
  a.href = dataUrl;
  a.download = 'poster_' + formatTimeFn(time) + '.' + ext;
  a.innerHTML = '<i class="ti ti-download" aria-hidden="true"></i> Download';
  div.appendChild(img);
  div.appendChild(tspan);
  div.appendChild(a);
  grid.appendChild(div);
}

function clearThumbs() {
  document.getElementById('thumbGrid').innerHTML = '';
  thumbCount = 0;
}

function formatTime(s) {
  if (!isFinite(s)) return '0s';
  const m = Math.floor(s / 60);
  const sec = (s % 60).toFixed(1);
  return m > 0 ? m + ':' + (sec < 10 ? '0' : '') + sec : sec + 's';
}

function formatTimeFn(s) {
  return s.toFixed(1).replace('.', '_');
}
</script>
