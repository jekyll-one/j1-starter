the two official plugins, `videojs-playlist` (logic) and `videojs-playlist-ui`

re-renders the menu in place — *without a single page reload*.

    (function () {
    'use strict';

    // Namespacing -------------------------------------------------------------
    var ROOT_ID   = 'vjs_playlist_test';
    var PLAYER_ID = 'vjt_player';

    // Pinned, mutually-compatible CDN versions (the same set the standalone
    // tool was verified against). Video.js core is only fetched when the J1
    // `videojs` resource has not already exposed window.videojs.
    var CDN = {
        coreCss: 'https://vjs.zencdn.net/8.10.0/video-js.css',
        coreJs:  'https://vjs.zencdn.net/8.10.0/video.min.js',
        // Video.js 8-compatible plugin builds. The pre-8 releases (playlist 4.x,
        // playlist-ui 3.x) were ES5-transpiled and call Component.call(this) for
        // their super-constructor, which Video.js 8's native ES6 Component class
        // rejects with "Class constructor Component$1 cannot be invoked without 'new'".
        // playlist-ui 5.x ships a single videojs-playlist-ui.css (the separate
        // .vertical.css is gone; vertical/horizontal is now a modifier class the
        // plugin applies automatically).
        plUiCss: 'https://cdn.jsdelivr.net/npm/videojs-playlist-ui@5.0.0/dist/videojs-playlist-ui.css',
        plJs:    'https://cdn.jsdelivr.net/npm/videojs-playlist@5.2.0/dist/videojs-playlist.min.js',
        plUiJs:  'https://cdn.jsdelivr.net/npm/videojs-playlist-ui@5.0.0/dist/videojs-playlist-ui.min.js'
    };
    var PLUGIN_VERSIONS = 'playlist 5.2.0 · playlist-ui 5.0.0';

    // Tiny asset helpers ------------------------------------------------------
    function byId(id) { return document.getElementById(id); }

    function addCss(href) {
        if (document.querySelector('link[href="' + href + '"]')) return;
        var l = document.createElement('link');
        l.rel = 'stylesheet';
        l.href = href;
        document.head.appendChild(l);
    }

    function addJs(src) {
        return new Promise(function (resolve, reject) {
        var existing = document.querySelector('script[src="' + src + '"]');
        if (existing) {
            if (existing.getAttribute('data-loaded') === '1') { resolve(); return; }
            existing.addEventListener('load', function () { resolve(); });
            existing.addEventListener('error', function () { reject(new Error('load failed: ' + src)); });
            return;
        }
        var s = document.createElement('script');
        s.src = src;
        s.async = false; // preserve execution order
        s.onload = function () { s.setAttribute('data-loaded', '1'); resolve(); };
        s.onerror = function () { reject(new Error('load failed: ' + src)); };
        document.head.appendChild(s);
        });
    }

    // Dependency loader
    // -------------------------------------------------------------------------

    // 1) Reuse the J1-provided Video.js if present, else pull core from the CDN.
    //    The base skin CSS is always ensured so the player is styled regardless
    //    of what the surrounding page ships.
    function ensureVideojs() {
        addCss(CDN.coreCss);
        if (window.videojs) return Promise.resolve();
        return addJs(CDN.coreJs);
    }

    // 2) Add the two official playlist plugins + the vertical playlist-ui CSS,
    //    skipping any that are already registered on this Video.js instance.
    function ensurePlugins() {
        addCss(CDN.plUiCss);
        var vjs = window.videojs;
        var hasPlugin = function (name) {
        return !!(vjs && typeof vjs.getPlugin === 'function' && vjs.getPlugin(name));
        };

        var chain = Promise.resolve();
        if (!hasPlugin('playlist'))   { chain = chain.then(function () { return addJs(CDN.plJs); }); }
        if (!hasPlugin('playlistUi')) { chain = chain.then(function () { return addJs(CDN.plUiJs); }); }
        return chain;
    }

    function boot() {
        ensureVideojs()
        .then(ensurePlugins)
        .then(initTool)
        .catch(function (err) {
            console.error('[vjs-playlist-test] init failed:', err);
            var v = byId('vjt_versions');
            if (v) { v.textContent = 'init failed — see console'; }
        });
    }

    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', boot);
    } else {
        boot();
    }

    // The test tool itself ----------------------------------------------------
    function initTool() {
        var videojs = window.videojs;
        var root = byId(ROOT_ID);
        if (!videojs || !root || !byId(PLAYER_ID)) { return; }

        // ---- Demo playlist --------------------------------------------------
        // Each item: standard Video.js source list + the fields videojs-playlist-ui
        // reads to build the menu (name, description, duration, thumbnail/poster).
        function demoPlaylist() {

            },

            sources: [ {  src: 'https://ia801903.us.archive.org/32/items/BigBuckBunny_328/BigBuckBunny_512kb.mp4', type: 'video/mp4' } ]
            },
            {
            name:         'Oceans',
            description:  'Test clip hosted by the Video.js Zencoder CDN.',

        // ---- Logging --------------------------------------------------------
        var logEl = byId('vjt_log');
        function log(msg, cls) {
        var time = new Date().toLocaleTimeString();
        var line = document.createElement('div');
        line.innerHTML = '<span class="t">' + time + '</span>  ' +
                        '<span class="' + (cls || 'info') + '">' + msg + '</span>';
        logEl.appendChild(line);
        logEl.scrollTop = logEl.scrollHeight;
        }
        byId('vjt_btnClearLog').onclick = function () { logEl.innerHTML = ''; };

        // ---- Initialise the player ------------------------------------------
        var player = videojs(PLAYER_ID, {
        fluid: false,
        playbackRates: [ 0.5, 1, 1.5, 2 ]
        });

        byId('vjt_versions').textContent = 'video.js ' + videojs.VERSION + ' · ' + PLUGIN_VERSIONS;

        var currentAA = 'off';

        player.ready(function () {

        // 1) Load the playlist via the official plugin.
        player.playlist(demoPlaylist());

        // 2) Build the clickable menu inside <div class="vjs-playlist">.
        player.playlistUi();

        // 3) Default behaviour.
        player.playlist.autoadvance(null); // disabled until the user picks one
        player.playlist.repeat(false);

        refreshState();
        bindControls();
        bindEvents();
        log('playlist loaded: ' + player.playlist().length + ' items', 'ok');

        // ---- Read live state into the panel ---------------------------------
        function refreshState() {
        var list = player.playlist() || [];
        var idx  = player.playlist.currentItem(); // returns the current index
        var item = (idx > -1 && list[idx]) ? list[idx] : null;

        byId('vjt_stIdx').textContent    = idx;
        byId('vjt_stName').textContent   = item ? (item.name || '(unnamed)') : '–';
        byId('vjt_stLen').textContent    = list.length;
        byId('vjt_stNext').textContent   = player.playlist.nextIndex();
        byId('vjt_stPrev').textContent   = player.playlist.previousIndex();
        byId('vjt_stRepeat').textContent = player.playlist.repeat();
        byId('vjt_stAA').textContent     = currentAA;

        var src = (item && item.sources && item.sources[0]) ? item.sources[0].src : '–';
        byId('vjt_stSrc').textContent    = src;

        byId('vjt_jumpIdx').max = Math.max(0, list.length - 1);
        }

        // ---- Wire up the control panel --------------------------------------
        function bindControls() {
        byId('vjt_btnFirst').onclick = function () { player.playlist.first(); };
        byId('vjt_btnPrev').onclick  = function () { player.playlist.previous(); };
        byId('vjt_btnNext').onclick  = function () { player.playlist.next(); };
        byId('vjt_btnLast').onclick  = function () { player.playlist.last(); };

        byId('vjt_btnJump').onclick = function () {
            var i = parseInt(byId('vjt_jumpIdx').value, 10);
            var len = player.playlist().length;
            if (isNaN(i) || i < 0 || i >= len) { log('jump: index out of range (0–' + (len - 1) + ')', 'warn'); return; }
            player.playlist.currentItem(i); // setter form jumps to the index
        };

        byId('vjt_btnPlay').onclick  = function () { player.play(); };
        byId('vjt_btnPause').onclick = function () { player.pause(); };

        byId('vjt_repeat').onchange = function () {
            player.playlist.repeat(this.checked);
            log('repeat(' + this.checked + ')', 'e');
            refreshState();
        };

        Array.prototype.forEach.call(document.getElementsByName('vjt_aa'), function (r) {
            r.onchange = function () {
            currentAA = this.value;
            if (this.value === 'off') {
                player.playlist.autoadvance(null);
            } else {
                player.playlist.autoadvance(parseInt(this.value, 10));
            }
            log('autoadvance(' + (this.value === 'off' ? 'null' : this.value) + ')', 'e');
            refreshState();
            };
        });

        byId('vjt_btnAdd').onclick = function () {
            var url = byId('vjt_addUrl').value.trim();
            if (!url) { log('add: enter a source URL first', 'warn'); return; }
            var list = player.playlist();
            list.push({
            name: 'Custom ' + (list.length + 1),
            description: url,
            sources: [ { src: url, type: guessType(url) } ]
            });
            player.playlist(list); // re-set rebuilds the menu, no reload
            byId('vjt_addUrl').value = '';
            log('added item #' + (list.length - 1) + ': ' + url, 'ok');
            refreshState();
        };

        byId('vjt_btnRemove').onclick = function () {
            var list = player.playlist();
            var idx  = player.playlist.currentItem();
            if (list.length <= 1) { log('remove: refusing to empty the playlist', 'warn'); return; }
            var removed = list.splice(idx, 1)[0];
            player.playlist(list);
            log('removed item: ' + (removed.name || idx), 'warn');
            refreshState();
        };

        byId('vjt_btnShuffle').onclick = function () {
            var list = player.playlist().slice();
            for (var i = list.length - 1; i > 0; i--) {
